const player = document.getElementById('player');
let posX = 0;
let posY = 0;
const step = 10;

function movePlayer(event) {
    switch (event.key) {
        case 'ArrowUp':
            posY = Math.max(0, posY - step);
            break;
        case 'ArrowDown':
            posY = Math.min(350, posY + step); // 400 - player height
            break;
        case 'ArrowLeft':
            posX = Math.max(0, posX - step);
            break;
        case 'ArrowRight':
            posX = Math.min(550, posX + step); // 600 - player width
            break;
    }
    player.style.transform = `translate(${posX}px, ${posY}px)`;
}

document.addEventListener('keydown', movePlayer);
